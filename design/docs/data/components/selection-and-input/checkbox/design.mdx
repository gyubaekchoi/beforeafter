<SectionVariants
  components={['FormField', 'FlexBox', 'FormControl', 'FormLabel', 'Checkbox']}
  icons={[]}
  render={(value) => {
      const tight = value['Tight'] === 'True';
      const size = value['Size'].toLowerCase();
      const bold = value['Bold'] === 'True';

      const gap = tight ? '10px' : '8px';

      return `
        <FormField sx={{ width: '75%' }} flexDirection="row" gap="${gap}" alignItems="center" justifyContent="center">
          <FormControl>
            <Checkbox
              size="${size}"
              bold={${bold}}
              tight={${tight}}
            />
          </FormControl>
          <FormLabel>Checkbox</FormLabel>
        </FormField>
      `;
    }}
  variants={[
    {
      key: 'Size',
      defaultValue: 'Medium',
      options: [
        { label: 'Small', value: {} },
        { label: 'Medium', value: {} }
      ]
    },
    {
      key: 'Tight',
      options: [
        { label: 'False', value: {} },
        { label: 'True', value: {} }
      ]
    },
    {
      key: 'Bold',
      defaultValue: 'False',
      options: [
        { label: 'True', value: {} },
        { label: 'False', value: {} }
      ]
    }
  ]}
/>
